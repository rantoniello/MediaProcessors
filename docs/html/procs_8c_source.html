369 void procs_close(procs_ctx_t **ref_procs_ctx)
370 {
371  procs_ctx_t *procs_ctx= NULL;
372  LOG_CTX_INIT(NULL);
373  LOGD(">>%s\n", __FUNCTION__); //comment-me
374 
375  if(ref_procs_ctx== NULL)
376  return;
377 
378  if((procs_ctx= *ref_procs_ctx)!= NULL) {
379  int proc_id, i;
380  LOG_CTX_SET(procs_ctx->log_ctx);
381 
382  /* First of all release all the processors (note that for deleting
383  * the processors we need the processor IF type to be still available).
384  */
385  LOCK_PROCS_CTX_API(procs_ctx);
386  for(proc_id= 0; proc_id< PROCS_MAX_NUM_PROC_INSTANCES; proc_id++) {
387  LOGD("unregistering proc with Id.: %d\n", proc_id); //comment-me
388  proc_unregister(procs_ctx, proc_id, LOG_CTX_GET());
389  }
390  UNLOCK_PROCS_CTX_API(procs_ctx);
391 
392  /* Module's instance API mutual exclusion lock */
393  ASSERT(pthread_mutex_destroy(&procs_ctx->api_mutex)== 0);
394 
395  /* Array listing the registered processor instances */
396  for(proc_id= 0; proc_id< PROCS_MAX_NUM_PROC_INSTANCES; proc_id++) {
397  procs_reg_elem_t *procs_reg_elem=
398  &procs_ctx->procs_reg_elem_array[proc_id];
399 
400  ASSERT(pthread_mutex_destroy(&procs_reg_elem->api_mutex)== 0);
401 
402  for(i= 0; i< PROC_IO_NUM; i++)
403  fair_lock_close(&procs_reg_elem->fair_lock_io_array[i]);
404  }
405 
406  /* Release module's instance context structure */
407  free(procs_ctx);
408  *ref_procs_ctx= NULL;
409  }
410  LOGD("<<%s\n", __FUNCTION__); //comment-me
411 }

525 static int register_proc_if(const proc_if_t *proc_if, log_ctx_t *log_ctx)
526 {
527  llist_t *n;
528  int ret_code, end_code= STAT_ERROR;
529  proc_if_t *proc_if_cpy= NULL;
530  LOG_CTX_INIT(log_ctx);
531 
532  /* Check arguments */
533  CHECK_DO(procs_module_ctx!= NULL, return STAT_ERROR);
534  CHECK_DO(proc_if!= NULL, return STAT_ERROR);
535 
536  /* Check that module API critical section is locked */
537  ret_code= pthread_mutex_trylock(&procs_module_ctx->module_api_mutex);
538  CHECK_DO(ret_code== EBUSY, return STAT_ERROR);
539 
540  /* Check if processor is already register with given "name" */
541  for(n= procs_module_ctx->proc_if_llist; n!= NULL; n= n->next) {
542  proc_if_t *proc_if_nth= (proc_if_t*)n->data;
543  CHECK_DO(proc_if_nth!= NULL, continue);
544  if(strcmp(proc_if_nth->proc_name, proc_if->proc_name)== 0) {
545  end_code= STAT_ECONFLICT;
546  goto end;
547  }
548  }
549 
550  /* Allocate a copy of the processor type in the list */
551  proc_if_cpy= proc_if_dup(proc_if);
552  //LOGV("Registering processor with name: '%s'\n",
553  // proc_if_cpy->proc_name); //comment-me
554  CHECK_DO(proc_if_cpy!= NULL, goto end);
555  ret_code= llist_push(&procs_module_ctx->proc_if_llist, proc_if_cpy);
556  CHECK_DO(ret_code== STAT_SUCCESS, goto end);
557 
558  end_code= STAT_SUCCESS;
559 end:
560  if(end_code!= STAT_SUCCESS)
561  proc_if_release(&proc_if_cpy);
562  return end_code;
563 }

565 static int unregister_proc_if(const char *proc_name, log_ctx_t *log_ctx)
566 {
567  llist_t **ref_n;
568  int ret_code;
569  LOG_CTX_INIT(log_ctx);
570 
571  /* Check arguments */
572  CHECK_DO(procs_module_ctx!= NULL, return STAT_ERROR);
573  CHECK_DO(proc_name!= NULL, return STAT_ERROR);
574 
575  /* Check that module instance critical section is locked */
576  ret_code= pthread_mutex_trylock(&procs_module_ctx->module_api_mutex);
577  CHECK_DO(ret_code== EBUSY, return STAT_ERROR);
578 
579  for(ref_n= &procs_module_ctx->proc_if_llist; (*ref_n)!= NULL;
580  ref_n= &((*ref_n)->next)) {
581  proc_if_t *proc_if_nth= (proc_if_t*)(*ref_n)->data;
582  CHECK_DO(proc_if_nth!= NULL, continue);
583 
584  if(strcmp(proc_if_nth->proc_name, proc_name)== 0) { // Node found
585  void *node;
586 
587  node= llist_pop(ref_n);
588  ASSERT(node!= NULL && node== (void*)proc_if_nth);
589 
590  /* Once that node register was popped (and thus not accessible
591  * by any concurrent thread), release corresponding context
592  * structure.
593  */
594  //LOGD("Unregistering processor '%s' succeed\n",
595  // proc_if_nth->proc_name); // comment-me
596  proc_if_release(&proc_if_nth);
597  ASSERT(proc_if_nth== NULL);
598  return STAT_SUCCESS;
599  }
600  }
601  return STAT_ENOTFOUND;
602 }

688  ret_code= pthread_mutex_trylock(&procs_ctx->api_mutex);
689  CHECK_DO(ret_code== EBUSY, return STAT_ERROR);
690 
691  *ref_rest_str= NULL;
692 
693  /* JSON structure is as follows:
694  * {
695  * "procs":
696  * [
697  * {
698  * "proc_id":number,
699  * "proc_name":string,
700  * "links":
701  * [
702  * {"rel":"self", "href":string}
703  * ]
704  * },
705  * ....
706  * ]
707  * }
708  */
709 
710  /* Create cJSON tree-root object */
711  cjson_rest= cJSON_CreateObject();
712  CHECK_DO(cjson_rest!= NULL, goto end);
713 
714  /* Create and attach 'PROCS' array */
715  cjson_procs= cJSON_CreateArray();
716  CHECK_DO(cjson_procs!= NULL, goto end);
717  cJSON_AddItemToObject(cjson_rest, "procs", cjson_procs);
718 
719  for(i= 0; i< PROCS_MAX_NUM_PROC_INSTANCES; i++) {
720  const char *proc_name;
721  const proc_if_t *proc_if;
722  register int proc_instance_index;
723  cJSON *cjson_proc, *cjson_links, *cjson_link;
724  proc_ctx_t *proc_ctx= NULL;
725  procs_reg_elem_t *procs_reg_elem=
726  &procs_ctx->procs_reg_elem_array[i];
727 
728  if((proc_ctx= procs_reg_elem->proc_ctx)== NULL)
729  continue;
730 
731  proc_instance_index= proc_ctx->proc_instance_index;
732  CHECK_DO(proc_instance_index== i, continue);
733 
734  cjson_proc= cJSON_CreateObject();
735  CHECK_DO(cjson_proc!= NULL, goto end);
736  cJSON_AddItemToArray(cjson_procs, cjson_proc);
737 
738  /* 'proc_id' */
739  cjson_aux= cJSON_CreateNumber((double)proc_instance_index);
740  CHECK_DO(cjson_aux!= NULL, goto end);
741  cJSON_AddItemToObject(cjson_proc, "proc_id", cjson_aux);
742 
743  /* 'proc_name' */
744  proc_if= proc_ctx->proc_if;
745  CHECK_DO(proc_if!= NULL, continue);
746  proc_name= proc_if->proc_name;
747  CHECK_DO(proc_name!= NULL, continue);
748  cjson_aux= cJSON_CreateString(proc_name);
749  CHECK_DO(cjson_aux!= NULL, goto end);
750  cJSON_AddItemToObject(cjson_proc, "proc_name", cjson_aux);
751 
752  /* 'links' */
753  cjson_links= cJSON_CreateArray();
754  CHECK_DO(cjson_links!= NULL, goto end);
755  cJSON_AddItemToObject(cjson_proc, "links", cjson_links);
756 
757  cjson_link= cJSON_CreateObject();
758  CHECK_DO(cjson_link!= NULL, goto end);
759  cJSON_AddItemToArray(cjson_links, cjson_link);
760 
761  cjson_aux= cJSON_CreateString("self");
762  CHECK_DO(cjson_aux!= NULL, goto end);
763  cJSON_AddItemToObject(cjson_link, "rel", cjson_aux);
764 
765  snprintf(href, sizeof(href), PROCS_URL_BASE_PATH"%d.json",
766  proc_instance_index);
767  cjson_aux= cJSON_CreateString(href);
768  CHECK_DO(cjson_aux!= NULL, goto end);
769  cJSON_AddItemToObject(cjson_link, "href", cjson_aux);
770  }
771 
772  /* Print cJSON structure data to char string */
773  *ref_rest_str= CJSON_PRINT(cjson_rest);
774  CHECK_DO(*ref_rest_str!= NULL && strlen(*ref_rest_str)> 0, goto end);
775 
776  end_code= STAT_SUCCESS;
777 end:
778  if(cjson_rest!= NULL)
779  cJSON_Delete(cjson_rest);
780  return end_code;

783 static int proc_register(procs_ctx_t *procs_ctx, const char *proc_name,
784  const char *settings_str, log_ctx_t *log_ctx, int *ref_id, va_list arg)
785 {
786  procs_reg_elem_t *procs_reg_elem;
787  const proc_if_t *proc_if;
788  int proc_id, ret_code, end_code= STAT_ERROR;
789  proc_ctx_t *proc_ctx= NULL;
790  uint32_t fifo_ctx_maxsize[PROC_IO_NUM]= {PROCS_FIFO_SIZE, PROCS_FIFO_SIZE};
791  LOG_CTX_INIT(log_ctx);
792 
793  /* Check arguments */
794  CHECK_DO(procs_module_ctx!= NULL, return STAT_ERROR);
795  CHECK_DO(procs_ctx!= NULL, return STAT_ERROR);
796  CHECK_DO(proc_name!= NULL, return STAT_ERROR);
797  CHECK_DO(settings_str!= NULL, return STAT_ERROR);
798  // Note: argument 'log_ctx' is allowed to be NULL
799  CHECK_DO(ref_id!= NULL, return STAT_ERROR);
800 
801  *ref_id= -1; // Set to invalid (undefined) value
802 
803  /* Check that module instance critical section is locked */
804  ret_code= pthread_mutex_trylock(&procs_ctx->api_mutex);
805  CHECK_DO(ret_code== EBUSY, goto end);
806 
807  /* Get free slot where to register new processor.
808  * Note that working on a locked module instance API critical section
809  * guarantee that 'procs_reg_elem_array' is not accessed concurrently for
810  * registering/unregistering.
811  * Nevertheless, 'procs_reg_elem_array' is still accessible concurrently
812  * by module's i/o operations, until corresponding "fair-lock" is locked.
813  */
814  for(proc_id= 0; proc_id< PROCS_MAX_NUM_PROC_INSTANCES; proc_id++) {
815  if(procs_ctx->procs_reg_elem_array[proc_id].proc_ctx== NULL)
816  break;
817  }
818  if(proc_id>= PROCS_MAX_NUM_PROC_INSTANCES) {
819  LOGE("Maximum number of allowed processor instances exceeded\n");
820  end_code= STAT_ENOMEM;
821  goto end;
822  }
823 
824  procs_reg_elem= &procs_ctx->procs_reg_elem_array[proc_id];
825 
826  /* Get processor interface (lock module!) */
827  ASSERT(pthread_mutex_lock(&procs_module_ctx->module_api_mutex)== 0);
828  proc_if= get_proc_if_by_name(proc_name, LOG_CTX_GET());
829  ASSERT(pthread_mutex_unlock(&procs_module_ctx->module_api_mutex)== 0);
830  if(proc_if== NULL) {
831  end_code= STAT_ENOTFOUND;
832  goto end;
833  }
834 
835  /* Open processor */
836  proc_ctx= proc_open(proc_if, settings_str, proc_id, fifo_ctx_maxsize,
837  LOG_CTX_GET(), arg);
838  CHECK_DO(proc_ctx!= NULL, goto end);
839 
840  /* Register processor context structure.
841  * At this point we lock corresponding i/o critical section ("fair" lock),
842  * as this register slot is still accessible concurrently by processor
843  * API and i/o operations.
844  */
845  LOCK_PROCS_REG_ELEM_API(procs_ctx, procs_reg_elem, goto end);
846  fair_lock(procs_reg_elem->fair_lock_io_array[PROC_IPUT]);
847  fair_lock(procs_reg_elem->fair_lock_io_array[PROC_OPUT]);
848  procs_reg_elem->proc_ctx= proc_ctx;
849  fair_unlock(procs_reg_elem->fair_lock_io_array[PROC_IPUT]);
850  fair_unlock(procs_reg_elem->fair_lock_io_array[PROC_OPUT]);
851  UNLOCK_PROCS_REG_ELEM_API(procs_reg_elem);
852  proc_ctx= NULL; // Avoid double referencing
853 
854  *ref_id= proc_id;
855  end_code= STAT_SUCCESS;
856 end:
857  if(proc_ctx!= NULL) {
858  proc_close(&proc_ctx);
859  }
860  return end_code;
861 }

863 static int proc_unregister(procs_ctx_t *procs_ctx, int proc_id,
864  log_ctx_t *log_ctx)
865 {
866  procs_reg_elem_t *procs_reg_elem;
867  int ret_code;
868  proc_ctx_t *proc_ctx= NULL;
869  LOG_CTX_INIT(log_ctx);
870  LOGD(">>%s\n", __FUNCTION__); //comment-me
871 
872  /* Check arguments */
873  CHECK_DO(procs_module_ctx!= NULL, return STAT_ERROR);
874  CHECK_DO(procs_ctx!= NULL, return STAT_ERROR);
875  CHECK_DO(proc_id>= 0 && proc_id< PROCS_MAX_NUM_PROC_INSTANCES,
876  return STAT_ERROR);
877  // Note: argument 'log_ctx' is allowed to be NULL
878 
879  /* Check that module instance critical section is locked */
880  ret_code= pthread_mutex_trylock(&procs_ctx->api_mutex);
881  CHECK_DO(ret_code== EBUSY, return STAT_ERROR);
882 
883  procs_reg_elem= &procs_ctx->procs_reg_elem_array[proc_id];
884 
885  /* Fetch processor.
886  * Note that working on locked module's API critical section guarantee that
887  * 'procs_reg_elem_array' is not accessed concurrently for
888  * registering/unregistering.
889  */
890  proc_ctx= procs_reg_elem->proc_ctx;
891  if(proc_ctx== NULL)
892  return STAT_ENOTFOUND;
893  ASSERT(proc_ctx->proc_instance_index== proc_id);
894 
895  /* Unblock processor input/output FIFOs to be able to acquire i/o locks */
896  ret_code= proc_opt(proc_ctx, "PROC_UNBLOCK");
897  CHECK_DO(ret_code== STAT_SUCCESS, return STAT_ERROR);
898 
899  /* Lock processor API and i/o critical sections.
900  * Delete processor reference from array register.
901  */
902  LOCK_PROCS_REG_ELEM_API(procs_ctx, procs_reg_elem, return STAT_ERROR);
903  fair_lock(procs_reg_elem->fair_lock_io_array[PROC_IPUT]);
904  fair_lock(procs_reg_elem->fair_lock_io_array[PROC_OPUT]);
905  procs_reg_elem->proc_ctx= NULL;
906  fair_unlock(procs_reg_elem->fair_lock_io_array[PROC_IPUT]);
907  fair_unlock(procs_reg_elem->fair_lock_io_array[PROC_OPUT]);
908  UNLOCK_PROCS_REG_ELEM_API(procs_reg_elem);
909 
910  /* Once processor register was deleted (and thus not accessible
911  * by any concurrent thread performing i/o), release corresponding context
912  * structure.
913  */
914  proc_close(&proc_ctx);
915  ASSERT(proc_ctx== NULL);
916  LOGD("<<%s\n", __FUNCTION__); //comment-me
917  return STAT_SUCCESS;
918 }

920 static int procs_id_opt(procs_ctx_t *procs_ctx, const char *tag,
921  log_ctx_t *log_ctx, va_list arg)
922 {
923  procs_reg_elem_t *procs_reg_elem;
924  int end_code= STAT_ERROR, proc_id= -1;
925  int flag_procs_api_locked= 0, flag_proc_ctx_api_locked= 0;
926  proc_ctx_t *proc_ctx= NULL;
927  LOG_CTX_INIT(log_ctx);
928 
929  /* Check arguments */
930  CHECK_DO(procs_module_ctx!= NULL, return STAT_ERROR);
931  CHECK_DO(procs_ctx!= NULL, return STAT_ERROR);
932  CHECK_DO(tag!= NULL, return STAT_ERROR);
933 
934  /* Implementation note:
935  * We use a double lock technique to access processor instance:
936  * - Lock PROCS module instance API critical section (while we are
937  * accessing processors register);
938  * - Lock processor (PROC) API critical section;
939  * - Fetch processor using Id. argument;
940  * - If fetching succeed, unlock PROCS module instance API critical
941  * section (we are able to leave processors register critical section);
942  * - ... use processor instance to treat options ...
943  * - Unlock processor (PROC) API critical section.
944  */
945 
946  /* Lock PROCS module instance API critical section */
947  LOCK_PROCS_CTX_API(procs_ctx);
948  flag_procs_api_locked= 1;
949 
950  /* Lock processor (PROC) API critical section */
951  proc_id= va_arg(arg, int);
952  CHECK_DO(proc_id>= 0 && proc_id< PROCS_MAX_NUM_PROC_INSTANCES, goto end);
953  procs_reg_elem= &procs_ctx->procs_reg_elem_array[proc_id];
954  LOCK_PROCS_REG_ELEM_API(procs_ctx, procs_reg_elem, goto end);
955  flag_proc_ctx_api_locked= 1;
956 
957  /* Fetch processor */
958  proc_ctx= procs_id_opt_fetch_proc_ctx(procs_ctx, proc_id, tag, arg,
959  LOG_CTX_GET());
960  if(proc_ctx== NULL) {
961  end_code= STAT_ENOTFOUND;
962  goto end;
963  }
964  ASSERT(proc_ctx->proc_instance_index== proc_id); // sanity check
965 
966  /* Unlock PROCS module instance API critical section */
967  UNLOCK_PROCS_CTX_API(procs_ctx);
968  flag_procs_api_locked= 0;
969 
970  /* Process options */
971  if(TAG_IS("PROCS_ID_GET")) {
972  end_code= procs_id_get(procs_reg_elem, proc_ctx, LOG_CTX_GET(),
973  (void**)va_arg(arg, char**));
974  } else if(TAG_IS("PROCS_ID_PUT")) {
975  end_code= proc_opt(proc_ctx, "PROC_PUT", va_arg(arg, const char*));
976  } else if(TAG_IS("PROCS_ID_UNBLOCK")) {
977  end_code= proc_opt(proc_ctx, "PROC_UNBLOCK");
978  } else {
979  // We assume is a private option for the processor
980  end_code= proc_vopt(proc_ctx, tag, arg);
981  }
982 
983 end:
984  /* Check critical sections and unlock if applicable */
985  if(flag_procs_api_locked!= 0)
986  UNLOCK_PROCS_CTX_API(procs_ctx);
987  if(flag_proc_ctx_api_locked!= 0)
988  UNLOCK_PROCS_REG_ELEM_API(procs_reg_elem);
989  return end_code;
990 }

1066 static proc_ctx_t* procs_id_opt_fetch_proc_ctx(procs_ctx_t *procs_ctx,
1067  int proc_id, const char *tag, va_list arg, log_ctx_t *log_ctx)
1068 {
1069  va_list arg_cpy, va_list_empty;
1070  procs_reg_elem_t *procs_reg_elem;
1071  int ret_code;
1072  int flag_is_query= 0; // 0-> JSON / 1->query string
1073  proc_ctx_t *proc_ctx_ret= NULL, *proc_ctx_curr= NULL, *proc_ctx_new= NULL;
1074  const char *settings_str_arg= NULL; // Do not release
1075  const proc_if_t *proc_if_curr= NULL, *proc_if_new= NULL; // Do not release
1076  const char *proc_name_curr= NULL; // Do not release
1077  char *proc_name_str= NULL;
1078  cJSON *cjson_rest_arg= NULL, *cjson_rest_curr= NULL;
1079  cJSON *cjson_aux= NULL; // Do not release
1080  char *settings_str_curr= NULL;
1081  uint32_t fifo_ctx_maxsize[PROC_IO_NUM]= {PROCS_FIFO_SIZE, PROCS_FIFO_SIZE};
1082  LOG_CTX_INIT(log_ctx);
1083 
1084  /* Check arguments */
1085  CHECK_DO(procs_ctx!= NULL, return NULL);
1086  CHECK_DO(proc_id>= 0 && proc_id< PROCS_MAX_NUM_PROC_INSTANCES, return NULL);
1087  CHECK_DO(tag!= NULL, return NULL);
1088  //arg nothing to check
1089  //log_ctx allowed to be NULL
1090 
1091  /* Get register element and current processor references */
1092  procs_reg_elem= &procs_ctx->procs_reg_elem_array[proc_id];
1093  proc_ctx_curr= procs_reg_elem->proc_ctx;
1094  CHECK_DO(proc_ctx_curr!= NULL, goto end);
1095 
1096  /* Check that module instance critical section is locked and processor API
1097  * level critical section is also locked ("double locked").
1098  */
1099  ret_code= pthread_mutex_trylock(&procs_ctx->api_mutex);
1100  CHECK_DO(ret_code== EBUSY, goto end);
1101  ret_code= pthread_mutex_trylock(&procs_reg_elem->api_mutex);
1102  CHECK_DO(ret_code== EBUSY, goto end);
1103 
1104  /* In the case we have a PUT operation request we may have to treat a
1105  * special case (code below). Otherwise, we're done.
1106  */
1107  if(!TAG_IS("PROCS_ID_PUT")) {
1108  proc_ctx_ret= proc_ctx_curr;
1109  goto end;
1110  }
1111 
1112  /* **** Treat possible special case in PUT operation ****
1113  * Check if a new processor type is requested (namely, check for a PUT
1114  * operation with a new processor name specified).
1115  */
1116 
1117  /* Copy variable list of arguments to avoid interfering with original.
1118  * Get settings string argument corresponding to the PUT operation.
1119  */
1120  va_copy(arg_cpy, arg);
1121  settings_str_arg= va_arg(arg_cpy, const char*);
1122  if(settings_str_arg== NULL || strlen(settings_str_arg)== 0) {
1123  // No new processor name requested, we're done.
1124  proc_ctx_ret= proc_ctx_curr;
1125  goto end;
1126  }
1127 
1128  /* Get current processor name */
1129  proc_if_curr= proc_ctx_curr->proc_if;
1130  CHECK_DO(proc_if_curr!= NULL, goto end);
1131  proc_name_curr= proc_if_curr->proc_name;
1132  CHECK_DO(proc_name_curr!= NULL && strlen(proc_name_curr)> 0, goto end);
1133 
1134  /* Guess string representation format (JSON-REST or Query) */
1135  //LOGV("'%s'\n", str); //comment-me
1136  flag_is_query= (settings_str_arg[0]=='{' &&
1137  settings_str_arg[strlen(settings_str_arg)-1]=='}')? 0: 1;
1138 
1139  /* Parse JSON or query string and check for 'proc_name' field */
1140  if(flag_is_query== 1) {
1141  proc_name_str= uri_parser_query_str_get_value("proc_name",
1142  settings_str_arg);
1143  } else {
1144  /* In the case string format is JSON-REST, parse to cJSON structure */
1145  cjson_rest_arg= cJSON_Parse(settings_str_arg);
1146  CHECK_DO(cjson_rest_arg!= NULL, goto end);
1147  cjson_aux= cJSON_GetObjectItem(cjson_rest_arg, "proc_name");
1148  if(cjson_aux!= NULL)
1149  proc_name_str= strdup(cjson_aux->valuestring);
1150  }
1151  if(proc_name_str== NULL) {
1152  // No new processor name requested, we're done.
1153  proc_ctx_ret= proc_ctx_curr;
1154  goto end;
1155  }
1156  //LOGV("'proc_name'= '%s' in PUT request\n", proc_name_str); //comment-me
1157 
1158  /* Check if processor name actually change or is the same as current */
1159  if(strcmp(proc_name_str, proc_name_curr)== 0) {
1160  // No new processor name requested, we're done.
1161  proc_ctx_ret= proc_ctx_curr;
1162  goto end;
1163  }
1164 
1165  /* Check and get processor interface (lock PROCS module API!) */
1166  ASSERT(pthread_mutex_lock(&procs_module_ctx->module_api_mutex)== 0);
1167  proc_if_new= get_proc_if_by_name(proc_name_str, LOG_CTX_GET());
1168  ASSERT(pthread_mutex_unlock(&procs_module_ctx->module_api_mutex)== 0);
1169  if(proc_if_new== NULL) {
1170  LOGE("New processor name specified '%s' is not registered.\n",
1171  proc_name_str);
1172  goto end;
1173  }
1174 
1175  /* **** At this point we have a valid new processor request **** */
1176  LOGW("Changing processor type from '%s' to '%s'\n",
1177  proc_if_curr->proc_mime? proc_if_curr->proc_mime: proc_name_curr,
1178  proc_if_new->proc_mime? proc_if_new->proc_mime: proc_name_str);
1179 
1180  /* Get processor's current settings */
1181  ret_code= proc_opt(proc_ctx_curr, "PROC_GET", PROC_IF_REST_FMT_CJSON,
1182  &cjson_rest_curr);
1183  CHECK_DO(ret_code== STAT_SUCCESS && cjson_rest_curr!= NULL, goto end);
1184  cjson_aux= cJSON_GetObjectItem(cjson_rest_curr, "settings");
1185  if(cjson_aux!= NULL) { // May have settings object or not...
1186  settings_str_curr= cJSON_PrintUnformatted(cjson_aux);
1187  } else {
1188  settings_str_curr= strdup("");
1189  }
1190  CHECK_DO(settings_str_curr!= NULL, goto end);
1191  //LOGV("Current processor setting: '%s'\n", settings_str_curr); //comment-me
1192 
1193  /* Open (instantiate) new processor passing current settings and Id.
1194  * Note that all the settings fields that do not apply to the new
1195  * processor type will be just ignored.
1196  */
1197  proc_ctx_new= proc_open(proc_if_new, settings_str_curr,
1198  proc_ctx_curr->proc_instance_index, fifo_ctx_maxsize,
1199  LOG_CTX_GET(), va_list_empty);
1200  CHECK_DO(proc_ctx_new!= NULL, goto end);
1201 
1202  /* Unblock current processor input/output FIFOs to be able to acquire
1203  * i/o locks next.
1204  */
1205  ret_code= proc_opt(proc_ctx_curr, "PROC_UNBLOCK");
1206  CHECK_DO(ret_code== STAT_SUCCESS, goto end);
1207 
1208  /* Register new processor context structure.
1209  * At this point we lock corresponding i/o critical section ("fair" lock),
1210  * as this register slot is still accessible concurrently by processor
1211  * API and i/o operations.
1212  */
1213  //LOCK_PROCS_REG_ELEM_API(...); //already
1214  fair_lock(procs_reg_elem->fair_lock_io_array[PROC_IPUT]);
1215  fair_lock(procs_reg_elem->fair_lock_io_array[PROC_OPUT]);
1216  procs_reg_elem->proc_ctx= proc_ctx_new;
