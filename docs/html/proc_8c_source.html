95 proc_ctx_t* proc_open(const proc_if_t *proc_if, const char *settings_str,
96  int proc_instance_index, uint32_t fifo_ctx_maxsize[PROC_IO_NUM],
97  log_ctx_t *log_ctx, va_list arg)
98 {
99  int i, ret_code, end_code= STAT_ERROR;
100  proc_ctx_t *proc_ctx= NULL;
101  fifo_elem_alloc_fxn_t fifo_elem_alloc_fxn= {0};
102  LOG_CTX_INIT(log_ctx);
103 
104  /* Check arguments */
105  CHECK_DO(proc_if!= NULL, return NULL);
106  CHECK_DO(settings_str!= NULL, return NULL);
107  CHECK_DO(fifo_ctx_maxsize!= NULL , return NULL);
108  // Note: 'log_ctx' is allowed to be NULL
109 
110  /* Check mandatory call-backs existence */
111  CHECK_DO(proc_if->open!= NULL, goto end);
112  CHECK_DO(proc_if->close!= NULL, goto end);
113  CHECK_DO(proc_if->process_frame!= NULL, goto end);
114 
115  /* Open (allocate) the specific processor (PROC) instance */
116  proc_ctx= proc_if->open(proc_if, settings_str, LOG_CTX_GET(), arg);
117  CHECK_DO(proc_ctx!= NULL, goto end);
118 
119  /* Set processor (PROC) interface context structure */
120  proc_ctx->proc_if= proc_if;
121 
122  /* Set PROC register index. */
123  proc_ctx->proc_instance_index= proc_instance_index;
124 
125  /* API mutual exclusion lock */
126  ret_code= pthread_mutex_init(&proc_ctx->api_mutex, NULL);
127  CHECK_DO(ret_code== 0, goto end);
128 
129  /* Set external LOG module */
130  proc_ctx->log_ctx= LOG_CTX_GET();
131 
132  /* Initialize input FIFO buffer */
133  fifo_elem_alloc_fxn.elem_ctx_dup=
134  proc_if->iput_fifo_elem_opaque_dup!= NULL?
135  (fifo_elem_ctx_dup_fxn_t*)proc_if->iput_fifo_elem_opaque_dup:
136  (fifo_elem_ctx_dup_fxn_t*)proc_frame_ctx_dup;
137  fifo_elem_alloc_fxn.elem_ctx_release=
138  proc_if->iput_fifo_elem_opaque_release!= NULL?
139  (fifo_elem_ctx_release_fxn_t*)
140  proc_if->iput_fifo_elem_opaque_release:
141  (fifo_elem_ctx_release_fxn_t*)proc_frame_ctx_release;
142 
143  proc_ctx->fifo_ctx_array[PROC_IPUT]= fifo_open(
144  fifo_ctx_maxsize[PROC_IPUT], 0, &fifo_elem_alloc_fxn);
145  CHECK_DO(proc_ctx->fifo_ctx_array[PROC_IPUT]!= NULL, goto end);
146 
147  /* Initialize output FIFO buffer */
148  fifo_elem_alloc_fxn.elem_ctx_dup=
149  proc_if->oput_fifo_elem_opaque_dup!= NULL?
150  (fifo_elem_ctx_dup_fxn_t*)proc_if->oput_fifo_elem_opaque_dup:
151  (fifo_elem_ctx_dup_fxn_t*)proc_frame_ctx_dup;
152  fifo_elem_alloc_fxn.elem_ctx_release=
153  (fifo_elem_ctx_release_fxn_t*)proc_frame_ctx_release;
154  proc_ctx->fifo_ctx_array[PROC_OPUT]= fifo_open(
155  fifo_ctx_maxsize[PROC_OPUT], 0, &fifo_elem_alloc_fxn);
156  CHECK_DO(proc_ctx->fifo_ctx_array[PROC_OPUT]!= NULL, goto end);
157 
158  /* Initialize input/output fair-locks */
159  for(i= 0; i< PROC_IO_NUM; i++) {
160  fair_lock_t *fair_lock= fair_lock_open();
161  CHECK_DO(fair_lock!= NULL, goto end);
162  proc_ctx->fair_lock_io_array[i]= fair_lock;
163  }
164 
165  /* Initialize input/output MUTEX for bitrate statistics related */
166  ret_code= pthread_mutex_init(&proc_ctx->acc_io_bits_mutex[PROC_IPUT], NULL);
167  CHECK_DO(ret_code== 0, goto end);
168  ret_code= pthread_mutex_init(&proc_ctx->acc_io_bits_mutex[PROC_OPUT], NULL);
169  CHECK_DO(ret_code== 0, goto end);
170 
171  /* Initialize array registering input presentation time-stamps (PTS's) */
172  proc_ctx->iput_pts_array_idx= 0;
173  memset(proc_ctx->iput_pts_array, -1, sizeof(proc_ctx->iput_pts_array));
174 
175  /* Initialize latency measurement related variables */
176  proc_ctx->acc_latency_nsec= 0;
177  proc_ctx->acc_latency_cnt= 0;
178  ret_code= pthread_mutex_init(&proc_ctx->latency_mutex, NULL);
179  CHECK_DO(ret_code== 0, goto end);
180 
181  /* At last, launch PROC thread */
182  proc_ctx->flag_exit= 0;
183  proc_ctx->start_routine= (const void*(*)(void*))proc_thr;
184  ret_code= pthread_create(&proc_ctx->proc_thread, NULL, proc_thr, proc_ctx);
185  CHECK_DO(ret_code== 0, goto end);
186 
187  end_code= STAT_SUCCESS;
188 end:
189  if(end_code!= STAT_SUCCESS)
190  proc_if->close(&proc_ctx);
191  return proc_ctx;
192 }

194 void proc_close(proc_ctx_t **ref_proc_ctx)
195 {
196  proc_ctx_t *proc_ctx= NULL;
197  LOG_CTX_INIT(NULL);
198  LOGD(">>%s\n", __FUNCTION__); //comment-me
199 
200  if(ref_proc_ctx== NULL)
201  return;
202 
203  if((proc_ctx= *ref_proc_ctx)!= NULL) {
204  const proc_if_t *proc_if;
205  void *thread_end_code= NULL;
206  LOG_CTX_SET(proc_ctx->log_ctx);
207 
208  /* Join processing thread first
209  * - set flag to notify we are exiting processing;
210  * - unlock input and output FIFO's;
211  * - join thread.
212  */
213  proc_ctx->flag_exit= 1;
214  fifo_set_blocking_mode(proc_ctx->fifo_ctx_array[PROC_IPUT], 0);
215  fifo_set_blocking_mode(proc_ctx->fifo_ctx_array[PROC_OPUT], 0);
216  LOGD("Waiting thread to join... "); // comment-me
217  pthread_join(proc_ctx->proc_thread, &thread_end_code);
218  if(thread_end_code!= NULL) {
219  ASSERT(*((int*)thread_end_code)== STAT_SUCCESS);
220  free(thread_end_code);
221  thread_end_code= NULL;
222  }
223  LOGD("joined O.K.\n"); // comment-me
224 
225  /* Release API mutual exclusion lock */
226  ASSERT(pthread_mutex_destroy(&proc_ctx->api_mutex)== 0);
227 
228  /* Release input and output FIFO's */
229  fifo_close(&proc_ctx->fifo_ctx_array[PROC_IPUT]);
230  fifo_close(&proc_ctx->fifo_ctx_array[PROC_OPUT]);
231 
232  /* Release input/output fair locks */
233  fair_lock_close(&proc_ctx->fair_lock_io_array[PROC_IPUT]);
234  fair_lock_close(&proc_ctx->fair_lock_io_array[PROC_OPUT]);
235 
236  /* Release input/output MUTEX for bitrate statistics related */
237  ASSERT(pthread_mutex_destroy(&proc_ctx->acc_io_bits_mutex[PROC_IPUT])
238  == 0);
239  ASSERT(pthread_mutex_destroy(&proc_ctx->acc_io_bits_mutex[PROC_OPUT])
240  == 0);
241 
242  /* Release latency measurement related variables */
243  ASSERT(pthread_mutex_destroy(&proc_ctx->latency_mutex)== 0);
244 
245  /* Close the specific PROC instance */
246  CHECK_DO((proc_if= proc_ctx->proc_if)!= NULL, return); // sanity check
247  CHECK_DO(proc_if->close!= NULL, return); // sanity check
248  proc_if->close(ref_proc_ctx);
249  }
250  LOGD("<<%s\n", __FUNCTION__); //comment-me
251 }

279 int proc_recv_frame(proc_ctx_t *proc_ctx,
280  proc_frame_ctx_t **ref_proc_frame_ctx)
281 {
282  int ret_code, end_code= STAT_ERROR;
283  size_t fifo_elem_size= 0;
284  proc_frame_ctx_t *proc_frame_ctx= NULL; // Do not release
285  LOG_CTX_INIT(NULL);
286 
287  /* Check arguments */
288  CHECK_DO(proc_ctx!= NULL, return STAT_ERROR);
289  CHECK_DO(ref_proc_frame_ctx!= NULL, return STAT_ERROR);
290 
291  fair_lock(proc_ctx->fair_lock_io_array[PROC_OPUT]);
292 
293  LOG_CTX_SET(proc_ctx->log_ctx);
294 
295  *ref_proc_frame_ctx= NULL;
296 
297  /* Read a frame from the output FIFO */
298  ret_code= fifo_get(proc_ctx->fifo_ctx_array[PROC_OPUT],
299  (void**)ref_proc_frame_ctx, &fifo_elem_size);
300  if(ret_code!= STAT_SUCCESS) {
301  end_code= ret_code;
302  goto end;
303  }
304 
305  /* Update processor's output bitrate statistics
306  * (note we are "receiving frame *from* the processor").
307  * For most processors implementation (specially for encoders and
308  * decoders), measuring traffic at this point would be precise.
309  * Nevertheless, for certain processors, as is the case of multiplexers,
310  * this function ('proc_recv_frame()') is not used thus input bitrate
311  * should be measured at some other point of the specific implementation
312  * (e.g. when sending data to an INET socket).
313  */
314  if((proc_frame_ctx= *ref_proc_frame_ctx)!= NULL) {
315  register int i;
316  register uint32_t byte_cnt_oput, bit_cnt_oput;
317  register size_t width;
318  pthread_mutex_t *acc_io_bits_mutex_p=
319  &proc_ctx->acc_io_bits_mutex[PROC_OPUT];
320 
321  /* Get frame size (we "unroll" for the most common cases) */
322  byte_cnt_oput= (proc_frame_ctx->width[0]* proc_frame_ctx->height[0])+
323  (proc_frame_ctx->width[1]* proc_frame_ctx->height[1])+
324  (proc_frame_ctx->width[2]* proc_frame_ctx->height[2]);
325  // We do the rest in a loop...
326  for(i= 3; (width= proc_frame_ctx->width[i])> 0 &&
327  i< PROC_FRAME_NUM_DATA_POINTERS; i++)
328  byte_cnt_oput+= width* proc_frame_ctx->height[i];
329 
330  /* Update currently accumulated bit value */
331  bit_cnt_oput= (byte_cnt_oput<< 3);
332  ASSERT(pthread_mutex_lock(acc_io_bits_mutex_p)== 0);
333  proc_ctx->acc_io_bits[PROC_OPUT]+= bit_cnt_oput;
334  ASSERT(pthread_mutex_unlock(acc_io_bits_mutex_p)== 0);
335  }
336 
337  end_code= STAT_SUCCESS;
338 end:
339  fair_unlock(proc_ctx->fair_lock_io_array[PROC_OPUT]);
340  if(end_code!= STAT_SUCCESS)
341  proc_frame_ctx_release(ref_proc_frame_ctx);
342  return end_code;
343 }

395  }
396  }
397 
398  ASSERT(pthread_mutex_unlock(&proc_ctx->api_mutex)== 0);
399  return end_code;

402 void proc_acc_latency_measure(proc_ctx_t *proc_ctx,
403  const int64_t oput_frame_pts)
404 {
405  register int i, idx;
406  LOG_CTX_INIT(NULL);
407 
408  /* Check arguments */
409  CHECK_DO(proc_ctx!= NULL, return);
410  if(oput_frame_pts== 0)
411  return;
412 
413  LOG_CTX_SET(proc_ctx->log_ctx);
414 
415  /* Parse input circular PTS register.
416  * Note that we do not care if this register is modified
417  * concurrently in function 'proc_send_frame_stats_iput_pts()'...
418  * this measurement is still valid.
419  */
420  for(i= 0, idx= proc_ctx->iput_pts_array_idx; i< IPUT_PTS_ARRAY_SIZE;
421  i++, idx= (idx+ 1)% IPUT_PTS_ARRAY_SIZE) {
422  int64_t pts_iput= proc_ctx->iput_pts_array[IPUT_PTS_VAL][idx];
423 
424  if(pts_iput== oput_frame_pts) {
425  register int ret_code;
426  register int64_t curr_nsec, iput_nsec;
427  struct timespec monotime_curr= {0};
428 
429  ret_code= clock_gettime(CLOCK_MONOTONIC, &monotime_curr);
430  CHECK_DO(ret_code== 0, return);
431  curr_nsec= (int64_t)monotime_curr.tv_sec*1000000000+
432  (int64_t)monotime_curr.tv_nsec;
433  iput_nsec= proc_ctx->iput_pts_array[IPUT_PTS_STC_VAL][idx];
434  if(curr_nsec> iput_nsec) {
435  pthread_mutex_t *latency_mutex_p= &proc_ctx->latency_mutex;
436  ASSERT((pthread_mutex_lock(latency_mutex_p))== 0);
437  proc_ctx->acc_latency_nsec+= curr_nsec- iput_nsec;
438  proc_ctx->acc_latency_cnt++;
439  //LOGV("acc_latency_nsec= %"PRId64" (count: %d)\n",
440  // proc_ctx->acc_latency_nsec,
441  // proc_ctx->acc_latency_cnt); //comment-me
442  ASSERT((pthread_mutex_unlock(latency_mutex_p))== 0);
443  }
444  return;
445  }
446  }
447  return;
448 }

450 static int procs_id_get(proc_ctx_t *proc_ctx, log_ctx_t *log_ctx,
451  proc_if_rest_fmt_t rest_fmt, void **ref_reponse)
452 {
453  const proc_if_t *proc_if;
454  uint64_t flag_proc_features;
455  int ret_code, end_code= STAT_ERROR;
456  cJSON *cjson_rest= NULL, *cjson_aux= NULL;
457  int (*rest_get)(proc_ctx_t *proc_ctx, proc_if_rest_fmt_t rest_fmt,
458  void **ref_reponse)= NULL;
459  LOG_CTX_INIT(log_ctx);
460 
461  /* Check arguments */
462  CHECK_DO(proc_ctx!= NULL, return STAT_ERROR);
463  //log_ctx allowed to be NULL
464  CHECK_DO(rest_fmt>= 0 && rest_fmt< PROC_IF_REST_FMT_ENUM_MAX,
465  return STAT_ERROR);
466  CHECK_DO(ref_reponse!= NULL, return STAT_ERROR);
467 
468  *ref_reponse= NULL;
469 
470  /* Check that processor API critical section is locked */
471  ret_code= pthread_mutex_trylock(&proc_ctx->api_mutex);
472  CHECK_DO(ret_code== EBUSY, goto end);
473 
474  /* Get required variables from PROC interface structure */
475  proc_if= proc_ctx->proc_if;
476  CHECK_DO(proc_if!= NULL, goto end);
477  rest_get= proc_if->rest_get;
478  flag_proc_features= proc_if->flag_proc_features;
479 
480  /* Check if GET function callback is implemented by specific processor */
481  if(rest_get== NULL) {
482  /* Nothing to do */
483  end_code= STAT_ENOTFOUND;
484  goto end;
485  }
486 
487  /* GET processor's REST response */
488  ret_code= rest_get(proc_ctx, PROC_IF_REST_FMT_CJSON, (void**)&cjson_rest);
489  if(ret_code!= STAT_SUCCESS) {
490  end_code= ret_code;
491  goto end;
492  }
493  CHECK_DO(cjson_rest!= NULL, goto end);
494 
495  /* **** Add some REST elements at top ****
496  * We do a little HACK to insert elements at top as cJSON library does not
497  * support it natively -it always insert at the bottom-
498  * We do this at the risk of braking in a future library version, as we
499  * base current solution on the internal implementation of function
500  * 'cJSON_AddItemToObject()' -may change in future-.
501  */
502 
503  if(flag_proc_features&PROC_FEATURE_LATSTATS) {
504  /* 'latency_avg_usec' */
505  cjson_aux= cJSON_CreateNumber((double)proc_ctx->latency_avg_usec);
506  CHECK_DO(cjson_aux!= NULL, goto end);
507  // Hack of 'cJSON_AddItemToObject(cjson_rest, "latency_avg_usec",
508  // cjson_aux);':
509  cjson_aux->string= (char*)strdup("latency_avg_usec");
510  cjson_aux->type|= cJSON_StringIsConst;
511  //cJSON_AddItemToArray(cjson_rest, cjson_aux);
512  cJSON_InsertItemInArray(cjson_rest, 0, cjson_aux); // Insert at top
513  cjson_aux->type&= ~cJSON_StringIsConst;
514  }
515 
516  /* Format response to be returned */
517  switch(rest_fmt) {
518  case PROC_IF_REST_FMT_CHAR:
519  /* Print cJSON structure data to char string */
520  *ref_reponse= (void*)CJSON_PRINT(cjson_rest);
521  CHECK_DO(*ref_reponse!= NULL && strlen((char*)*ref_reponse)> 0,
522  goto end);
523  break;
524  case PROC_IF_REST_FMT_CJSON:
525  *ref_reponse= (void*)cjson_rest;
526  cjson_rest= NULL; // Avoid double referencing
527  break;
528  default:
529  goto end;
530  }
531 
532  end_code= STAT_SUCCESS;
533 end:
534  if(cjson_rest!= NULL)
535  cJSON_Delete(cjson_rest);
536  return end_code;
537 }

568  flag_proc_features= proc_if->flag_proc_features;
569 
570  while(proc_ctx->flag_exit== 0) {
571  register int ret_code;
572 
573  /* Check if i/o statistics are used by this processor */
574  if(flag_proc_features& PROC_FEATURE_IOSTATS) {
575  register uint32_t bit_counter_iput, bit_counter_oput;
576  pthread_mutex_t *acc_io_bits_mutex_iput_p=
577  &proc_ctx->acc_io_bits_mutex[PROC_IPUT];
578  pthread_mutex_t *acc_io_bits_mutex_oput_p=
579  &proc_ctx->acc_io_bits_mutex[PROC_OPUT];
580 
581  /* Note that this is a periodic loop executed once per second
582  * (Thus, 'bitrate' is given in bits per second)
583  */
584  ASSERT(pthread_mutex_lock(acc_io_bits_mutex_iput_p)== 0);
585  bit_counter_iput= proc_ctx->acc_io_bits[PROC_IPUT];
586  proc_ctx->acc_io_bits[PROC_IPUT]= 0;
587  ASSERT(pthread_mutex_unlock(acc_io_bits_mutex_iput_p)== 0);
588  proc_ctx->bitrate[PROC_IPUT]= bit_counter_iput;
589 
590  ASSERT(pthread_mutex_lock(acc_io_bits_mutex_oput_p)== 0);
591  bit_counter_oput= proc_ctx->acc_io_bits[PROC_OPUT];
592  proc_ctx->acc_io_bits[PROC_OPUT]= 0;
593  ASSERT(pthread_mutex_unlock(acc_io_bits_mutex_oput_p)== 0);
594  proc_ctx->bitrate[PROC_OPUT]= bit_counter_oput;
595  }
596 
597  /* Check if latency statistics are used by this processor */
598  if(flag_proc_features&PROC_FEATURE_LATSTATS) {
599  register int acc_latency_cnt;
600  register int64_t acc_latency_usec= 0;
601  pthread_mutex_t *latency_mutex_p= &proc_ctx->latency_mutex;
602 
603  ASSERT((pthread_mutex_lock(latency_mutex_p))== 0);
604  acc_latency_cnt= proc_ctx->acc_latency_cnt;
605  proc_ctx->acc_latency_cnt= 0;
606  if(acc_latency_cnt> 0)
607  acc_latency_usec= proc_ctx->acc_latency_nsec/ acc_latency_cnt;
608  proc_ctx->acc_latency_nsec= 0;
609  ASSERT((pthread_mutex_unlock(latency_mutex_p))== 0);
610  acc_latency_usec/= 1000; // convert nsec-> usec
611  proc_ctx->latency_avg_usec= acc_latency_usec;
612  //LOGV("----> Average latency (usec/sec)= %"PRId64"\n",
613  // acc_latency_usec); //comment-me
614  if(acc_latency_usec> proc_ctx->latency_max_usec)
615  proc_ctx->latency_max_usec= acc_latency_usec;
616  if(acc_latency_usec> 0 &&
617  acc_latency_usec< proc_ctx->latency_min_usec)
618  proc_ctx->latency_min_usec= acc_latency_usec;
619  }
620 
621  /* Sleep given time (interruptible by external thread) */
622  ret_code= interr_usleep(interr_usleep_ctx,
623  PROC_STATS_THR_MEASURE_PERIOD_USECS);
624  ASSERT(ret_code== STAT_SUCCESS || ret_code== STAT_EINTR);
625  }
626 
627  *ref_end_code= STAT_SUCCESS;
628 end:
629  return (void*)ref_end_code;
630 }

632 static void* proc_thr(void *t)
633 {
634  const proc_if_t *proc_if;
635  pthread_t stats_thread;
636  proc_ctx_t* proc_ctx= (proc_ctx_t*)t;
637  int ret_code, *ref_end_code= NULL;
638  int (*process_frame)(proc_ctx_t*, fifo_ctx_t*, fifo_ctx_t*)= NULL;
639  fifo_ctx_t *iput_fifo_ctx= NULL, *oput_fifo_ctx= NULL;
640  void *thread_end_code= NULL;
641  interr_usleep_ctx_t *interr_usleep_ctx= NULL;
642  proc_stats_thr_arg_t proc_stats_thr_arg= {0};
643  LOG_CTX_INIT(NULL);
644 
645  /* Allocate return context; initialize to a default 'ERROR' value */
646  ref_end_code= (int*)malloc(sizeof(int));
647  CHECK_DO(ref_end_code!= NULL, return NULL);
648  *ref_end_code= STAT_ERROR;
649 
650  /* Check arguments */
651  CHECK_DO(proc_ctx!= NULL, goto end);
652 
653  LOG_CTX_SET(proc_ctx->log_ctx);
654 
655  /* Get PROC processing callback */
656  proc_if= proc_ctx->proc_if;
657  CHECK_DO(proc_if!= NULL, goto end);
658  process_frame= proc_if->process_frame;
659  CHECK_DO(process_frame!= NULL, goto end);
660 
661  /* Get input/output FIFO buffers */
662  iput_fifo_ctx= proc_ctx->fifo_ctx_array[PROC_IPUT];
663  oput_fifo_ctx= proc_ctx->fifo_ctx_array[PROC_OPUT];
664  CHECK_DO(iput_fifo_ctx!= NULL && oput_fifo_ctx!= NULL, goto end);
665 
666  /* Launch periodical statistics computing thread
667  * (e.g. for computing processor's input/output bitrate statistics):
668  * - Instantiate (open) an interruptible usleep module instance;
669  * - Launch statistic thread passing corresponding argument structure.
670  */
671  interr_usleep_ctx= interr_usleep_open();
672  CHECK_DO(interr_usleep_ctx!= NULL, goto end);
673  proc_stats_thr_arg.proc_ctx= proc_ctx;
674  proc_stats_thr_arg.interr_usleep_ctx= interr_usleep_ctx;
675  ret_code= pthread_create(&stats_thread, NULL, proc_stats_thr,
676  (void*)&proc_stats_thr_arg);
677  CHECK_DO(ret_code== 0, goto end);
678 
679  /* Run processing thread */
680  while(proc_ctx->flag_exit== 0) {
681  ret_code= process_frame(proc_ctx, iput_fifo_ctx, oput_fifo_ctx);
682  if(ret_code== STAT_EOF)
683  proc_ctx->flag_exit= 1;
684  else if(ret_code!= STAT_SUCCESS)
685  schedule(); // Avoid CPU-consuming closed loops
686  }
687 
688  *ref_end_code= STAT_SUCCESS;
689 end:
690  /* Join periodical statistics thread:
691  * - Unlock interruptible usleep module instance;
692  * - Join the statistics thread;
693  * - Release (close) the interruptible usleep module instance.
694  */
695  interr_usleep_unblock(interr_usleep_ctx);
696  pthread_join(stats_thread, &thread_end_code);
697  if(thread_end_code!= NULL) {
698  ASSERT(*((int*)thread_end_code)== STAT_SUCCESS);
699  free(thread_end_code);
700  thread_end_code= NULL;
701  }
702  interr_usleep_close(&interr_usleep_ctx);
703  return (void*)ref_end_code;
704 }

759 static void proc_send_frame_stats_io(proc_ctx_t *proc_ctx,
760  const proc_frame_ctx_t *proc_frame_ctx)
761 {
762  const proc_if_t *proc_if;
763  register uint64_t flag_proc_features;
764  register uint32_t byte_cnt_iput, bit_cnt_iput;
765  register size_t width;
766  register int i;
767  pthread_mutex_t *acc_io_bits_mutex_p= NULL;
768  LOG_CTX_INIT(NULL);
769 
770  /* Check arguments */
771  CHECK_DO(proc_ctx!= NULL, return);
772  CHECK_DO(proc_frame_ctx!= NULL, return);
773 
774  LOG_CTX_SET(proc_ctx->log_ctx);
775 
776  /* Get required variables from PROC interface structure */
777  proc_if= proc_ctx->proc_if;
778  CHECK_DO(proc_if!= NULL, return);
779  flag_proc_features= proc_if->flag_proc_features;
780 
781  /* Check if i/o statistics are used by this processor */
782  if(!(flag_proc_features& PROC_FEATURE_IOSTATS))
783  return;
784 
785  /* Update processor's input bitrate statistics
786  * (note we are "sending frame *into* the processor").
787  * For most processors implementation (specially for encoders and
788  * decoders), measuring traffic at this point would be precise.
789  * Nevertheless, for certain processors, as is the case of demultiplexers,
790  * this function ('proc_send_frame()') is not used thus input bitrate
791  * should be measured at some other point of the specific implementation
792  * (e.g. when receiving data from an INET socket).
793  */
794  acc_io_bits_mutex_p= &proc_ctx->acc_io_bits_mutex[PROC_IPUT];
795 
796  /* Get frame size (we "unroll" for the most common cases) */
797  byte_cnt_iput= (proc_frame_ctx->width[0]* proc_frame_ctx->height[0])+
798  (proc_frame_ctx->width[1]* proc_frame_ctx->height[1])+
799  (proc_frame_ctx->width[2]* proc_frame_ctx->height[2]);
800  // We do the rest in a loop...
801  for(i= 3; (width= proc_frame_ctx->width[i])> 0 &&
802  i< PROC_FRAME_NUM_DATA_POINTERS; i++)
803  byte_cnt_iput+= width* proc_frame_ctx->height[i];
804 
805  /* Update currently accumulated bit value */
806  bit_cnt_iput= (byte_cnt_iput<< 3);
807  ASSERT(pthread_mutex_lock(acc_io_bits_mutex_p)== 0);
808  proc_ctx->acc_io_bits[PROC_IPUT]+= bit_cnt_iput;
809  ASSERT(pthread_mutex_unlock(acc_io_bits_mutex_p)== 0);
810 
811  return;
812 }
