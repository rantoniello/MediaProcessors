272 static int ffmpeg_mlhe_enc_process_frame(proc_ctx_t *proc_ctx,
273  fifo_ctx_t* iput_fifo_ctx, fifo_ctx_t* oput_fifo_ctx)
274 {
275  int ret_code, end_code= STAT_ERROR;
276  ffmpeg_mlhe_enc_ctx_t *ffmpeg_mlhe_enc_ctx= NULL; // Do not release
277  ffmpeg_video_enc_ctx_t *ffmpeg_video_enc_ctx= NULL; // Do not release
278  AVFrame *avframe_iput= NULL;
279  size_t fifo_elem_size= 0;
280  LOG_CTX_INIT(NULL);
281 
282  /* Check arguments */
283  CHECK_DO(proc_ctx!= NULL, return STAT_ERROR);
284  CHECK_DO(iput_fifo_ctx!= NULL, return STAT_ERROR);
285  CHECK_DO(oput_fifo_ctx!= NULL, return STAT_ERROR);
286 
287  LOG_CTX_SET(proc_ctx->log_ctx);
288 
289  /* Get FFmpeg video encoder context */
290  ffmpeg_mlhe_enc_ctx= (ffmpeg_mlhe_enc_ctx_t*)proc_ctx;
291  ffmpeg_video_enc_ctx= &ffmpeg_mlhe_enc_ctx->ffmpeg_video_enc_ctx;
292 
293  /* Get input frame from FIFO buffer */
294  ret_code= fifo_get(iput_fifo_ctx, (void**)&avframe_iput, &fifo_elem_size);
295  CHECK_DO(ret_code== STAT_SUCCESS || ret_code== STAT_EAGAIN, goto end);
296  if(ret_code== STAT_EAGAIN) {
297  /* This means FIFO was unblocked, just go out with EOF status */
298  end_code= STAT_EOF;
299  goto end;
300  }
301 
302  /* Encode frame */
303  ret_code= ffmpeg_video_enc_frame(ffmpeg_video_enc_ctx, avframe_iput,
304  oput_fifo_ctx, LOG_CTX_GET());
305  CHECK_DO(ret_code== STAT_SUCCESS || ret_code== STAT_EAGAIN, goto end);
306 
307  end_code= STAT_SUCCESS;
308 end:
309  if(avframe_iput!= NULL)
310  av_frame_free(&avframe_iput);
311  return end_code;
312 }

371  LOG_CTX_INIT(NULL);
372 
373  /* Check arguments */
374  CHECK_DO(proc_ctx!= NULL, return STAT_ERROR);
375  CHECK_DO(rest_fmt< PROC_IF_REST_FMT_ENUM_MAX, return STAT_ERROR);
376  CHECK_DO(ref_reponse!= NULL, return STAT_ERROR);
377 
378  LOG_CTX_SET(proc_ctx->log_ctx);
379 
380  *ref_reponse= NULL;
381 
382  /* Create cJSON tree root object */
383  cjson_rest= cJSON_CreateObject();
384  CHECK_DO(cjson_rest!= NULL, goto end);
385 
386  /* JSON string to be returned:
387  * {
388  * "settings":
389  * {
390  * ...
391  * },
392  * ... // Reserved for future use
393  * }
394  */
395 
396  /* Get FFmpeg video encoder settings contexts */
397  ffmpeg_mlhe_enc_ctx= (ffmpeg_mlhe_enc_ctx_t*)proc_ctx;
398  ffmpeg_mlhe_enc_settings_ctx=

407 
408  /* GET specific MLHE video encoder settings */
409  // Reserved for future use: attach to 'cjson_settings' (should be != NULL)
410 
411  /* Attach settings object to REST response */
412  cJSON_AddItemToObject(cjson_rest, "settings", cjson_settings);
413  cjson_settings= NULL; // Attached; avoid double referencing
414 
415  /* **** Attach data to REST response **** */

418  avcodecctx= ffmpeg_video_enc_ctx->avcodecctx;
419  CHECK_DO(avcodecctx!= NULL, goto end);
420 
421  // Reserved for future use
422  /* Example:
423  * cjson_aux= cJSON_CreateNumber((double)avcodecctx->var1);

425  * cJSON_AddItemToObject(cjson_rest, "var1_name", cjson_aux);
426  */
427 
428  // Reserved for future use: set other data values here...
429 
430  /* Format response to be returned */
431  switch(rest_fmt) {
432  case PROC_IF_REST_FMT_CHAR:
433  /* Print cJSON structure data to char string */
434  *ref_reponse= (void*)CJSON_PRINT(cjson_rest);
435  CHECK_DO(*ref_reponse!= NULL && strlen((char*)*ref_reponse)> 0,
436  goto end);
437  break;
438  case PROC_IF_REST_FMT_CJSON:
439  *ref_reponse= (void*)cjson_rest;
440  cjson_rest= NULL; // Avoid double referencing
441  break;
442  default:
443  goto end;
444  }
445 
446  end_code= STAT_SUCCESS;
447 end:
448  if(cjson_settings!= NULL)
449  cJSON_Delete(cjson_settings);
450  if(cjson_rest!= NULL)
451  cJSON_Delete(cjson_rest);
452  return end_code;
453 }
